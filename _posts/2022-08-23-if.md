```python
# 판정체크 1
import random
com = random.randint(1, 3)
player = 0
player = input("입력 : ")
player = int(player)

# 데이터 타입 통일하는거 잊지말자
# 기왕이면 처음에 바꾸고 시작하자 나중에 계속 str(), int()이러는거 귀찮다

while player <= 0 or player > 3:
    player = input("입력 : ")
    player = int(player)

if player == 1:
    print("당신은 가위")
elif player == 2:
    print("당신은 바위")
elif player == 3:
    print("당신은 보")
else:
    print("당신은 기권")

if com == 1:
    print("컴퓨터는 가위")
elif com == 2:
    print("컴퓨터는 바위")
else:
    print("컴퓨터는 보")
    
if player == 1 and com == 1:
    print("무승부")
elif player == 2 and com == 2:
    print("무승부")
elif player == 3 and com == 3:
    print("무승부")
elif player == 1 and com == 3:
    print("승리")
elif player == 2 and com == 1:
    print("승리")
elif player == 3 and com == 2:
    print("승리")
elif player == 1 and com == 2:
    print("패배")
elif player == 2 and com == 3:
    print("패배")
elif player == 3 and com == 1:
    print("패배")
else:
    print("패배")
print("끝")
```
